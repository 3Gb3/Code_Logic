‚ûï Adicionando Elementos

1. append() - Adicionar no Final

frutas = ["ma√ß√£", "banana"]
print(f"Inicial: {frutas}")

frutas.append("uva")
print(f"Ap√≥s append: {frutas}")  # ['ma√ß√£', 'banana', 'uva']

# Adicionando m√∫ltiplos elementos

2. insert() - Adicionar em Posi√ß√£o Espec√≠fica

cores = ["azul", "verde", "amarelo"]
print(f"Inicial: {cores}")

cores.insert(1, "vermelho")  # Inserir na posi√ß√£o 1
print(f"Ap√≥s insert: {cores}")  # ['azul', 'vermelho', 'verde', 'amarelo']

cores.insert(0, "preto")  # Inserir no in√≠cio
print(f"No in√≠cio: {cores}")  # ['preto', 'azul', 'vermelho', 'verde', 'amarelo']

cores.insert(-1, "branco")  # Inserir antes do √∫ltimo
print(f"Antes do √∫ltimo: {cores}")  # ['preto', 'azul', 'vermelho', 'verde', 'branco', 'amarelo']

3. extend() - Adicionar M√∫ltiplos Elementos

numeros = [1, 2, 3]
print(f"Inicial: {numeros}")

# extend() adiciona cada elemento individualmente
numeros.extend([4, 5, 6])
print(f"Ap√≥s extend: {numeros}")  # [1, 2, 3, 4, 5, 6]

# Diferen√ßa entre append e extend
lista1 = [1, 2, 3]
lista2 = [1, 2, 3]

lista1.append([4, 5, 6])  # Adiciona a lista como um elemento
lista2.extend([4, 5, 6])  # Adiciona cada elemento da lista

print(f"append: {lista1}")  # [1, 2, 3, [4, 5, 6]]
print(f"extend: {lista2}")  # [1, 2, 3, 4, 5, 6]

‚ûñ Removendo Elementos

1. remove() - Remove Primeira Ocorr√™ncia

animais = ["gato", "cachorro", "gato", "p√°ssaro"]
print(f"Inicial: {animais}")

animais.remove("gato")  # Remove o primeiro "gato"
print(f"Ap√≥s remove: {animais}")  # ['cachorro', 'gato', 'p√°ssaro']

# Cuidado: remove() gera erro se elemento n√£o existir
# Sempre verificar se existe antes de remover

    animais.remove("elefante")
else:
    print("Elemento 'elefante' n√£o encontrado")

2. pop() - Remove e Retorna Elemento

pilha = [1, 2, 3, 4, 5]
print(f"Inicial: {pilha}")

# pop() sem par√¢metro remove o √∫ltimo elemento
ultimo = pilha.pop()
print(f"Removido: {ultimo}")  # 5
print(f"Pilha: {pilha}")  # [1, 2, 3, 4]

# pop() com √≠ndice remove elemento espec√≠fico
segundo = pilha.pop(1)  # Remove elemento no √≠ndice 1
print(f"Removido: {segundo}")  # 2
print(f"Pilha: {pilha}")  # [1, 3, 4]

# Usando pop() em fila (primeiro elemento)
primeiro = pilha.pop(0)
print(f"Primeiro removido: {primeiro}")  # 1
print(f"Fila: {pilha}")  # [3, 4]

3. clear() - Remove Todos os Elementos

numeros = [1, 2, 3, 4, 5]
print(f"Antes: {numeros}")

print(f"Tamanho: {len(numeros)}")  # 0

4. del - Remove por √çndice ou Slice

letras = ['a', 'b', 'c', 'd', 'e', 'f']
print(f"Inicial: {letras}")

# Remover elemento espec√≠fico
del letras[2]  # Remove 'c'
print(f"Ap√≥s del: {letras}")  # ['a', 'b', 'd', 'e', 'f']

# Remover slice
del letras[1:3]  # Remove 'b' e 'd'
print(f"Ap√≥s del slice: {letras}")  # ['a', 'e', 'f']

# Remover elementos alternados
del letras[::2]  # Remove √≠ndices 0, 2, 4...
print(f"Ap√≥s del alternados: {letras}")  # ['e']

üîç M√©todos de Busca

1. index() - Encontrar Posi√ß√£o

# Encontrar primeira ocorr√™ncia

# Busca com in√≠cio e fim

    posicao_kiwi = frutas.index(elemento_busca)
    print(f"Posi√ß√£o de '{elemento_busca}': {posicao_kiwi}")

    print(f"Posi√ß√£o de '{elemento_busca}': -1")  # -1 indica n√£o encontrado

2. count() - Contar Ocorr√™ncias

print(f"N√∫mero 2 aparece {count_2} vezes")  # 3

print(f"N√∫mero 6 aparece {count_6} vezes")  # 0

# Contagem de elementos √∫nicos

    contagem[elemento] = numeros.count(elemento)

üìä M√©todos de Organiza√ß√£o

1. sort() - Ordenar Lista

numeros = [3, 1, 4, 1, 5, 9, 2, 6]
print(f"Original: {numeros}")

# Ordena√ß√£o crescente
numeros.sort()
print(f"Crescente: {numeros}")  # [1, 1, 2, 3, 4, 5, 6, 9]

# Ordena√ß√£o decrescente
numeros.sort(reverse=True)
print(f"Decrescente: {numeros}")  # [9, 6, 5, 4, 3, 2, 1, 1]

# Ordena√ß√£o de strings
nomes = ["Carlos", "Ana", "Bruno", "Diana"]
nomes.sort()
print(f"Nomes ordenados: {nomes}")  # ['Ana', 'Bruno', 'Carlos', 'Diana']

# Ordena√ß√£o personalizada (por tamanho)
palavras = ["python", "java", "c", "javascript"]
palavras.sort(key=len)
print(f"Por tamanho: {palavras}")  # ['c', 'java', 'python', 'javascript']

2. reverse() - Inverter Lista

letras = ['a', 'b', 'c', 'd', 'e']
print(f"Original: {letras}")

letras.reverse()
print(f"Invertida: {letras}")  # ['e', 'd', 'c', 'b', 'a']

# Diferen√ßa entre reverse() e sort(reverse=True)
numeros1 = [3, 1, 4, 2]
numeros2 = [3, 1, 4, 2]

numeros1.reverse()  # Apenas inverte ordem atual
numeros2.sort(reverse=True)  # Ordena e depois inverte

print(f"reverse(): {numeros1}")  # [2, 4, 1, 3]
print(f"sort(reverse=True): {numeros2}")  # [4, 3, 2, 1]

3. copy() - Copiar Lista

original = [1, 2, 3, 4, 5]

# C√≥pia rasa (shallow copy)

print(f"C√≥pia: {copia}")

# Modificar c√≥pia n√£o afeta original

print(f"Ap√≥s modificar c√≥pia:")

print(f"C√≥pia: {copia}")  # [1, 2, 3, 4, 5, 6]

# CUIDADO: Atribui√ß√£o n√£o cria c√≥pia!

‚ö° M√©todos √öteis Combinados

1. Limpeza e Filtragem:

2. Remo√ß√£o Segura em Loop:
