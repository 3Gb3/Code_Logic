def remover_duplicatas(lista):
    return list(set(lista))

# Preservar ordem
def remover_duplicatas_ordenado(lista):
    visto = set()
    resultado = []
    for item in lista:
        if item not in visto:
            visto.add(item)
            resultado.append(item)
    return resultado
