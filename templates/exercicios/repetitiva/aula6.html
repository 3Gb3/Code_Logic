def eh_palindromo(palavra):
    # Remove espa√ßos e converte para min√∫scula
    palavra_limpa = palavra.replace(" ", "").lower()
    
    tamanho = len(palavra_limpa)
    
    for i in range(tamanho // 2):
        if palavra_limpa[i] != palavra_limpa[tamanho - 1 - i]:
            return False
    
    return True

def cifra_cesar(texto, deslocamento):
    resultado = ""
    
    for char in texto:
        if char.isalpha():
            # Determina se √© mai√∫scula ou min√∫scula
            if char.isupper():
                base = ord('A')
            else:
                base = ord('a')
            
            # Aplica o deslocamento
            char_num = ord(char) - base
            char_cifrado = (char_num + deslocamento) % 26
            novo_char = chr(char_cifrado + base)
            
            resultado += novo_char
        else:
            # Mant√©m caracteres n√£o-alfab√©ticos
            resultado += char
    
    return resultado

# Teste

deslocamento = 3

mensagem_cifrada = cifra_cesar(mensagem, deslocamento)
mensagem_decifrada = cifra_cesar(mensagem_cifrada, -deslocamento)

print(f"Original: {mensagem}")
print(f"Cifrada: {mensagem_cifrada}")
print(f"Decifrada: {mensagem_decifrada}")

2. Criando Bordas

def criar_borda(texto, char="*"):
    linhas = texto.split('\n')
    largura_max = max(len(linha) for linha in linhas)
    
    # Linha superior
    print(char * (largura_max + 4))
    
    # Linhas do texto
    for linha in linhas:
        espacos = largura_max - len(linha)
        print(f"{char} {linha}{' ' * espacos} {char}")
    
    # Linha inferior
    print(char * (largura_max + 4))

# Teste
criar_borda("Python\n√©\nincr√≠vel!")

# Sa√≠da:
# ***********
# * Python  *
# * √©       *
# * incr√≠vel! *

def buscar_palavra(texto, palavra):
    texto_lower = texto.lower()
    palavra_lower = palavra.lower()
    
    encontradas = []
    palavras = texto.split()
    
    for i, p in enumerate(palavras):
        if p.lower() == palavra_lower:
            encontradas.append(i)
    
    return encontradas

# Teste
