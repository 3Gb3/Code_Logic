    [5, 10, 15],
    [20, 25, 30]
]

for linha in valores:
    print(linha)

        valores[i][j] += 10

for linha in valores:
    print(linha)

üîÑ Zerar Elementos Negativos

saldos = [
    [100, -50, 200],
    [-30, 150, -80],
    [120, -10, 90]
]

print("Saldos originais:")
for linha in saldos:
    print(linha)

# Substituir negativos por zero
for i in range(len(saldos)):
    for j in range(len(saldos[i])):
        if saldos[i][j] < 0:
            saldos[i][j] = 0

print("\nSaldos ajustados (negativos = 0):")
for linha in saldos:
    print(linha)
