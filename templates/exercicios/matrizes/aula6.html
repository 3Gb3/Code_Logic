🎯 Objetivo

Aprender a transpor matrizes (trocar linhas por colunas).

🔄 O que é Transposição?

Transpor uma matriz significa transformar linhas em colunas e colunas em linhas.

# Matriz original 3x2
original = [
    [1, 2],
    [3, 4],
    [5, 6]
]

# Transposta será 2x3
# [1, 3, 5]
# [2, 4, 6]

📐 Transpor Matriz Quadrada

matriz = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
]

print("Matriz Original:")
for linha in matriz:
    print(linha)

# Criar matriz transposta
n = len(matriz)
transposta = []

for i in range(n):
    linha = []
    for j in range(n):
        linha.append(matriz[j][i])  # Inverter índices
    transposta.append(linha)

print("\nMatriz Transposta:")
for linha in transposta:
    print(linha)

📊 Transpor Matriz Retangular

# Matriz 3x2
matriz = [
    [1, 2],
    [3, 4],
    [5, 6]
]

print("Matriz Original (3x2):")
for linha in matriz:
    print(linha)

# Transposta será 2x3
linhas_original = len(matriz)
colunas_original = len(matriz[0])

transposta = []

for j in range(colunas_original):
    linha = []
    for i in range(linhas_original):
        linha.append(matriz[i][j])
    transposta.append(linha)

print("\nMatriz Transposta (2x3):")
for linha in transposta:
    print(linha)
