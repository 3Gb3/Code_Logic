# Criar matriz identidade 4x4
tamanho = 4
identidade = []

for i in range(tamanho):
    linha = []
    for j in range(tamanho):
        if i == j:
            linha.append(1)
        else:
            linha.append(0)
    identidade.append(linha)

print("Matriz Identidade 4x4:")
for linha in identidade:
    print(linha)

ğŸ“ Diagonal Principal

matriz = [
    [10, 2, 3, 4],
    [5, 20, 7, 8],
    [9, 10, 30, 12],
    [13, 14, 15, 40]
]

print("Matriz:")
for linha in matriz:
    print(linha)

# Extrair diagonal principal
print("\nDiagonal principal:")
for i in range(len(matriz)):
    print(matriz[i][i], end=" ")

# Soma da diagonal
soma_diagonal = 0
for i in range(len(matriz)):
    soma_diagonal += matriz[i][i]

print(f"\nSoma da diagonal: {soma_diagonal}")

ğŸ”º Matriz Triangular Superior

Elementos abaixo da diagonal sÃ£o zero.

# Criar matriz triangular superior
tamanho = 4
triangular = []

numero = 1
for i in range(tamanho):
    linha = []
    for j in range(tamanho):
        if j >= i:  # Diagonal e acima
            linha.append(numero)
            numero += 1
        else:  # Abaixo da diagonal
            linha.append(0)
    triangular.append(linha)

print("Matriz Triangular Superior:")
for linha in triangular:
    for elem in linha:
        print(f"{elem:3}", end=" ")
    print()

# Criar matriz triangular inferior

triangular = []

    linha = []
    for j in range(tamanho):

            linha.append(numero)
            numero += 1
        else:  # Acima da diagonal
            linha.append(0)
    triangular.append(linha)

print("Matriz Triangular Inferior:")
for linha in triangular:
    for elem in linha:
        print(f"{elem:3}", end=" ")
    print()
