Template de an√°lise:

n = int(input("Ordem? "))

# Ler matriz
matriz = []
for i in range(n):
    linha = []
    for j in range(n):
        v = float(input(f"[{i},{j}]: "))
        linha.append(v)
    matriz.append(linha)

# 1. Soma total e m√©dia
soma_total = 0
for i in range(n):
    for j in range(n):
        soma_total += matriz[i][j]

quantidade = n * n
media = soma_total / quantidade

print(f"SOMA TOTAL: {soma_total:.1f}")
print(f"MEDIA: {media:.1f}")

# 2. Maior e menor
maior = matriz[0][0]
menor = matriz[0][0]
for i in range(n):
    for j in range(n):
        if matriz[i][j] > maior:
            maior = matriz[i][j]
        if matriz[i][j] < menor:
            menor = matriz[i][j]

print(f"MAIOR: {maior:.1f}")
print(f"MENOR: {menor:.1f}")

# 3. Contagens
positivos = 0
negativos = 0
zeros = 0
for i in range(n):
    for j in range(n):
        if matriz[i][j] > 0:
            positivos += 1
        elif matriz[i][j] < 0:
            negativos += 1
        else:
            zeros += 1

print(f"POSITIVOS: {positivos}")
print(f"NEGATIVOS: {negativos}")
print(f"ZEROS: {zeros}")

# 4. Diagonal principal
print("DIAGONAL PRINCIPAL:")
for i in range(n):
    print(f"{matriz[i][i]:.1f} ", end="")
print()

# 5. Soma acima da diagonal
soma_acima = 0
for i in range(n):
    for j in range(n):
        if i < j:
            soma_acima += matriz[i][j]

print(f"SOMA ACIMA DIAGONAL: {soma_acima:.1f}")
